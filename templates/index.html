				def list_files_current_directory_w_condition(directory=None):
				    """ List files in either provided directory or current directory if none is provided
				        with conditions on file extension """
				    lst_files = []
				    if directory is None:
				        directory = os.getcwd()
				    for filename in os.listdir(directory):
				        if filename.endswith(".txt") or filename.endswith(".csv"):
				            lst_files.append(directory + filename)
				        else:
				            continue
				    return lst_files
